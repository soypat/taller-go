Terminación de programa

        



//+build non-slide OMIT

package main

import (
	"fmt"
	"os"
)


func main() {
    defer fmt.Println("Terminando el programa!")
    fmt.Println("Comenzando el programa!")
    // En contexto de main() termina ejecución llamando el defer stack de main()
    return
    // panic ejecuta el defer stack y devuelve control a la función circundante
    // esto ocurre recursivamente hasta terminar el programa o si se encuentra
    // con una llamada a recover()
    panic("Oh no! Un Error!")
    // Termina inmediatamente.  0: Todo OK  1-255: Hubo un error
    os.Exit(0)
}

//+build ignore OMIT

package main

import "fmt"


type Juego struct {
    Nombre  string
    Puntaje float32
    Año     uint16
}

// Primero va el receptor de método (j Juego)
func (j Juego) Resumen() string {
    return fmt.Sprintf("%s [%d] %0.1f/10",
        j.Nombre, j.Año, j.Puntaje)
}

func main() {
    p := Juego{"Dark Souls", 8.6, 2011}
    fmt.Println(p.Resumen())
}

Método por referencia

        
La única diferencia de cambiar el receptor a que sea 
referencia es que se puede modificar los valores de la misma estructura 
en el mismo método





//+build ignore OMIT

package main

import "fmt"

type Juego struct {
	Nombre  string
	Puntaje float32
	Año     uint16
}


func (j Juego) Resumen() string { return fmt.Sprintf("%s [%d] %0.1f/10", j.Nombre, j.Año, j.Puntaje) }

func (j *Juego) NuevoPuntaje(p float32) {
    j.Puntaje = p
}

func main() {
    p := Juego{"Counter Strike", 7.5, 2001}
    fmt.Println(p.Resumen())
    p.NuevoPuntaje(9)
    fmt.Println(p.Resumen())
}

Interface

        



//+build ignore OMIT

package main

import "fmt"

// INTEFACE_S OMIT

type Lister interface {
    Push(string)
    Pop() string
}

type StringList []string

func (a StringList) Push(s string) { a = append(a, s) }

func (a StringList) Pop() (x string) {
    if len(a) >= 1 {
        x, a = a[len(a)-1], a[:len(a)-1]
    }
    return
}

func main() {
    var s Lister
    s = StringList{"QWERTY"}
    s.Push("DVORAK")
    fmt.Println(s)
}

Closure error wrapping

        



func AlgunaFuncion() (mainError error) {
    defer func() {
        if mainError != nil {
            mainError = fmt.Errorf("in AlgunaFuncion: %s", mainError)
        }
    }()
    mainError = ImGonnaError("too bad")
    if mainError != nil {
        return mainError
    }
    return nil
}








//+build ignore OMIT

package main

import (
	"fmt"
)

// MAIN_S OMIT
func main() {

    fmt.Println(AlgunaFuncion())



}

// MAIN_E OMIT
// FUNC_S OMIT
func AlgunaFuncion() (mainError error) {
	// FPROG_S OMIT
	defer func() {
		if mainError != nil {
			mainError = fmt.Errorf("in AlgunaFuncion: %s", mainError)
		}
	}()
	mainError = ImGonnaError("too bad")
	if mainError != nil {
		return mainError
	}
	return nil
	// FPROG_E OMIT
}

// FUNC_E OMIT
func ImGonnaError(s string) error {
	return fmt.Errorf("ImGonnaError did not expect %q", s)
}
